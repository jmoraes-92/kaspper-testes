-- Criação do banco de dados

CREATE DATABASE oracamentoskaspper;

USE oracamentoskaspper;

-- Tabela 'usuarios'

CREATE TABLE usuarios (

    id_usuario BIGINT AUTO_INCREMENT PRIMARY KEY,

    data_criacao DATETIME NOT NULL,

    email VARCHAR(150) NOT NULL UNIQUE,

    nome VARCHAR(100) NOT NULL,

    perfil ENUM('Administrador', 'Operador', 'Cliente') NOT NULL,

    senha VARCHAR(255) NOT NULL

);

-- Tabela 'clientes'

CREATE TABLE clientes (

    id_cliente BIGINT AUTO_INCREMENT PRIMARY KEY,

    data_cadastro DATETIME NOT NULL,

    email VARCHAR(255) NOT NULL UNIQUE,

    empresa VARCHAR(255),

    nome VARCHAR(255) NOT NULL,

    telefone VARCHAR(255),

    id_usuario BIGINT NOT NULL,

    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE CASCADE ON UPDATE CASCADE

);

-- Tabela 'demandas'

CREATE TABLE demandas (

    id_demanda BIGINT AUTO_INCREMENT PRIMARY KEY,

    data_criacao DATETIME NOT NULL,

    descricao TEXT NOT NULL,

    prioridade ENUM('Alta', 'Média', 'Baixa') NOT NULL,

    status ENUM('Aberta', 'Em andamento', 'Concluída') NOT NULL,

    id_cliente BIGINT NOT NULL,

    FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente) ON DELETE CASCADE ON UPDATE CASCADE

);

-- Tabela 'tarefas'

CREATE TABLE tarefas (

    id_tarefa BIGINT AUTO_INCREMENT PRIMARY KEY,

    descricao VARCHAR(255) NOT NULL,

    prazo DATE NOT NULL,

    responsavel VARCHAR(255),

    status TINYINT NOT NULL COMMENT '0 = Pendente, 1 = Concluída',

    id_demanda BIGINT NOT NULL,

    FOREIGN KEY (id_demanda) REFERENCES demandas(id_demanda) ON DELETE CASCADE ON UPDATE CASCADE

);

-- Tabela 'orcamentos'

CREATE TABLE orcamentos (

    id_orcamento BIGINT AUTO_INCREMENT PRIMARY KEY,

    observacoes TEXT,

    data_geracao DATETIME NOT NULL,

    status ENUM('Pendente', 'Aprovado', 'Rejeitado') NOT NULL,

    valor DECIMAL(38, 2) NOT NULL,

    prazo_estimado INT,

    id_demanda BIGINT NOT NULL,

    FOREIGN KEY (id_demanda) REFERENCES demandas(id_demanda) ON DELETE CASCADE ON UPDATE CASCADE

);

-- Tabela 'notificacoes'

CREATE TABLE notificacoes (

    id_notificacao BIGINT AUTO_INCREMENT PRIMARY KEY,

    data_envio DATETIME NOT NULL,

    mensagem TEXT NOT NULL,

    visualizada BIT(1) NOT NULL DEFAULT 0,

    id_usuario BIGINT,

    id_cliente BIGINT,

    id_orcamento BIGINT,

    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE SET NULL ON UPDATE CASCADE,

    FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente) ON DELETE SET NULL ON UPDATE CASCADE,

    FOREIGN KEY (id_orcamento) REFERENCES orcamentos(id_orcamento) ON DELETE SET NULL ON UPDATE CASCADE

);

Tabelas:

  • Cada tabela foi definida com seus atributos e tipos de dados correspondentes ao diagrama.
  • As chaves primárias (PRIMARY KEY) são adicionadas em cada tabela.
  • As chaves estrangeiras (FOREIGN KEY) garantem os relacionamentos e utilizam ON DELETE CASCADE para manter a integridade referencial, exceto para notificações que usam SET NULL.

Enumerações e Constraints:

  • ENUM é usado para campos com valores fixos como perfil, prioridade, e status.
  • Campos NOT NULL garantem que valores obrigatórios sejam inseridos.

Relacionalidade:

  • Relacionamentos (1:N) entre tabelas são definidos com chaves estrangeiras.
